# 视频相关概念

  **分辨率**
  -视频画面的像素尺寸，横向像素 × 纵向像素。
  -1920*1080
  
  **帧率(FPS)**
  -每秒播放的画面帧数，单位 fps。
  -帧率越高，画面越流畅。
  
  **码率**
  -单位时间内视频数据流量，单位 kbps/Mbps。
  -码率越高，画面清晰度越高，文件体积越大。
  
  **像素格式**
  -YUV420
    -Y：亮度分量（决定画面明暗）
    -U/V：色度分量（决定色彩）

# 视频开发流程
## 视频简单采集流程
1.  isp服务初始化
2.  rkmedia初始化
3.  视频采集vi通道设置
4.  使能vi通道
5.  开启视频流
6.  创建线程
7.  线程将采集数据写入文件
8.  阻塞进程
9.  失能vi通道
10. 关闭isp服务

**示例代码**
 ```c
 #include <stdio.h>
 #include <assert.h>
 #include <fcntl.h>
 #include <getopt.h>
 #include <signal.h>
 #include <stdbool.h>
 #include <stdlib.h>
 #include <string.h>
 #include <sys/types.h>
 #include <time.h>
 #include <unistd.h>
 
 #include "common/sample_common.h"
 #include "rkmedia_api.h"
 
 FILE * fd = NULL;
 int count =0;
 static void *get_video(void *arg)
 {
     fd = fopen("test.yuv","w+");
     while(1)
     {
         count++;
         //视频采集
         MEDIA_BUFFER mb=NULL;
         mb = RK_MPI_SYS_GetMediaBuffer(RK_ID_VI,0,-1);
         printf("%d视频已采集\n",count);
         //写入文件
         if(count < 200)
         {
             fwrite(RK_MPI_MB_GetPtr(mb),RK_MPI_MB_GetSize(mb),1,fd);
         }
         RK_MPI_MB_ReleaseBuffer(mb);
         if(count == 200)
         {
             fclose(fd);
             break;
         }
         
     }
 }
 
 int main()
 {
     VI_CHN_ATTR_S vi_attr;
     pthread_t pd;
     //isp配置
     rk_aiq_working_mode_t hdr_mode = RK_AIQ_WORKING_MODE_NORMAL;
     SAMPLE_COMM_ISP_Init(0, hdr_mode, RK_FALSE, "/oem/etc/iqfiles/");
     SAMPLE_COMM_ISP_Run(0);
     SAMPLE_COMM_ISP_SetFrameRate(0, 30);
 
     //系统初始化
     RK_MPI_SYS_Init();
 
     //视频采集通道属性
     vi_attr.pcVideoNode = "rkispp_scale0";
     vi_attr.u32BufCnt = 5;
     vi_attr.u32Width = 1920;
     vi_attr.u32Height = 1080;
     vi_attr.enPixFmt = IMAGE_TYPE_NV12;
     vi_attr.enWorkMode = VI_WORK_MODE_NORMAL;
     vi_attr.enBufType = VI_CHN_BUF_TYPE_MMAP;
     RK_MPI_VI_SetChnAttr(0,0,&vi_attr);
 
     //使能vi通道
     RK_MPI_VI_EnableChn(0,0);

     //开启视频流
     RK_MPI_VI_StartStream(0,0);

     //创建线程视频采集
     pthread_create(&pd,NULL,get_video,NULL);
 
     printf("视频采集完毕\n");
     getchar();
     
     //失能vi通道
     RK_MPI_VI_DisableChn(0,0);
 
     return 0;
 }
 ```
## 视频采集+编码流程
1.  isp服务初始化
2.  rkmedia初始化
3.  视频采集vi通道设置
4.  使能vi通道
5.  视频编码venc通道设置
6.  绑定venc通道与vi通道 开启视频流
7.  注册venc回调函数
8.  回调函数对vi采集的视频进行编码
9.  进程阻塞
10. 解除绑定
11. 失能venc通道
12. 失能vi通道
13. 停止isp服务

**示例代码**
 ```c
 #include <stdio.h>
 #include <assert.h>

 #include <getopt.h>

 #include "rkmedia_api.h"
 
 FILE *fp =NULL;
 int count;
 //venc输出数据回调函数
  void my_back(MEDIA_BUFFER mb)
  {
     count++;
     printf("%d正在采集\n",count);
     if(count<200)
     {
         fwrite(RK_MPI_MB_GetPtr(mb),RK_MPI_MB_GetSize(mb),1,fp);
     }
     RK_MPI_MB_ReleaseBuffer(mb);
     if(count == 200)

  }
 
 int main()
 {
     //isp配置
     fp = fopen("test.h264","w+");
     rk_aiq_working_mode_t hdr_mode = RK_AIQ_WORKING_MODE_NORMAL;
     SAMPLE_COMM_ISP_Init(0, hdr_mode, RK_FALSE, "/oem/etc/iqfiles/");
     SAMPLE_COMM_ISP_Run(0);
     SAMPLE_COMM_ISP_SetFrameRate(0, 30);
 
     //系统初始化
     RK_MPI_SYS_Init();
 
     //视频采集通道属性
     VI_CHN_ATTR_S vi_attr;
     vi_attr.pcVideoNode = "rkispp_scale0";
     vi_attr.u32BufCnt = 5;
     vi_attr.u32Width = 1920;
     vi_attr.u32Height = 1080;
     vi_attr.enPixFmt = IMAGE_TYPE_NV12;
     vi_attr.enWorkMode = VI_WORK_MODE_NORMAL;
     vi_attr.enBufType = VI_CHN_BUF_TYPE_MMAP;
     RK_MPI_VI_SetChnAttr(0,0,&vi_attr);
 
     //使能vi通道
     RK_MPI_VI_EnableChn(0,0);
 
     //创建venc编码通道
     VENC_CHN_ATTR_S venc_attr;
     venc_attr.stVencAttr.enType = RK_CODEC_TYPE_H264;
     venc_attr.stVencAttr.imageType = IMAGE_TYPE_NV12;
     venc_attr.stVencAttr.u32VirWidth = 1920;
     venc_attr.stVencAttr.u32VirHeight = 1080;
     venc_attr.stVencAttr.u32Profile = 77;
     venc_attr.stVencAttr.u32PicWidth = 1920;
     venc_attr.stVencAttr.u32PicHeight = 1080;
     venc_attr.stVencAttr.enRotation = 0;
     venc_attr.stRcAttr.enRcMode =VENC_RC_MODE_H264CBR;
     venc_attr.stRcAttr.stH264Cbr.u32Gop =30;
     venc_attr.stRcAttr.stH264Cbr.u32SrcFrameRateNum =30;
     venc_attr.stRcAttr.stH264Cbr.u32SrcFrameRateDen = 1;
     venc_attr.stRcAttr.stH264Cbr.fr32DstFrameRateNum = 30;
     venc_attr.stRcAttr.stH264Cbr.fr32DstFrameRateDen = 1;

     //注册venc输出回调
     MPP_CHN_S venc_chn;
     venc_chn.enModId = RK_ID_VENC;
     venc_chn.s32DevId =0;
     venc_chn.s32ChnId =0;
     RK_MPI_SYS_RegisterOutCb(&venc_chn,my_back);
     
 
     //通道绑定
     MPP_CHN_S src_chn,dest_chn;
     src_chn.enModId = RK_ID_VI;
     src_chn.s32DevId =0;
     src_chn.s32ChnId =0;
     dest_chn.enModId = RK_ID_VENC;
     dest_chn.s32DevId =0;
     dest_chn.s32ChnId =0;
     RK_MPI_SYS_Bind(&src_chn,&dest_chn);
 
 
     printf("按任意键退出采集\n");
     getchar();
     //取消通道绑定
     RK_MPI_SYS_UnBind(&src_chn,&dest_chn);
     //销毁venc通道
     RK_MPI_VENC_DestroyChn(0);
     //失能vi通道
     RK_MPI_VI_DisableChn(0,0);
     //停止isp服务
     SAMPLE_COMM_ISP_Stop(0);
     return 0;
 }
 ```
## 视频采集+编码+图像绘制流程(opencv)
1.  isp服务初始化
2.  rkmedia初始化
3.  视频采集vi通道设置
4.  使能vi通道
5.  视频编码venc通道设置
6.  RGA图层通道设置
7.  绑定RGA通道与VI通道，开启视频流 
8.  创建线程
9.  线程使用opencv对vi采集视频进行绘制
10. 注册venc回调函数
11. 回调函数对RGA绘制后的视频进行编码
12. 进程阻塞
13. 解除绑定
14. 失能rga通道
15. 失能venc通道
16. 失能vi通道
17. 停止isp服务
